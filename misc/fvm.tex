\section{Navier-Stokes}
\[
\underbrace{\frac{\partial(\rho  \bm{U})}{\partial t}+\nabla \cdot (\rho \bm{U} \bm{U})}_{\mathclap{\text{inertial forces}}}=
- \nabla p+ \underbrace{\nabla \cdot (\mu((\nabla \bm{U})+(\nabla \bm{U})^T) - \frac{2}{3}\mu(\nabla \cdot \bm{U})I)}_{\mathclap{\text{viscous forces}}} + \rho g
\]
\section{Incompressible Navier-Stokes}
\begin{align*}
	\frac{\partial\bm{U}}{\partial t} + \nabla \cdot ( \bm{U} \bm{U}) &= g - \nabla p + \nabla \cdot (\mu \nabla \bm{U}) \\
	\nabla \cdot \bm{U} &= 0
\end{align*}
